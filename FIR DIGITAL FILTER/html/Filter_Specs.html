FIR Digital Filter

Filter Coefficients

den = 1;
num=[1 3 3 1 4];
Ts=2E-5;

% Original Sine Wave
t=linspace(0,1,4096);
x=2000*sin(2*pi*t)+2000;

% Transfer Function and Filtering System
H=tf(num,den);
sys = filt(num,den,Ts);

% Filtering the oringal sine wave

% Quick check if stable
isstable(sys)

ans =

  logical

   1
